function performSearch() {

    var ajaxRequest = simpleAjaxUtil_getXmlHttpObject();     
	
    if (ajaxRequest==null) {alert ("Browser does not support HTTP Request");return;}
	
	
	var empresa = document.getElementById('empresa').value;
	var concepto = document.getElementById('concepto').value;
	var criterio = document.getElementById('criterio').value;
	
	
	
	document.getElementById('output').innerHTML = "<center>Buscando ... <img src='img/esperar.gif'></center>";
	
    var url = 'cap.jsp';
	var parameters = 	"empresa=" + empresa + "&" + 
						"concepto=" + concepto + "&" + 
						"criterio=" + criterio;
	

    ajaxRequest.open("post", url, true);
    ajaxRequest.setRequestHeader("Content-Type","application/x-www-form-urlencoded");
    ajaxRequest.onreadystatechange=function(){
    
	    if(ajaxRequest.readyState == 4 && ajaxRequest.status==200){
	
			//alert('Response');

		    var response = ajaxRequest.responseText;
		    document.getElementById('output').innerHTML = response;
	      } 
		  /*else {
			  alert(ajaxRequest.readyState);
			  alert(ajaxRequest.status);
			}*/
    };    
    ajaxRequest.send(parameters);
}



</script>
